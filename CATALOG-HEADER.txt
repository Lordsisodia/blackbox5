# BLACKBOX5 Catalog Header Template
# Add this to the top of key files for AI discovery

"""
BLACKBOX5_CATALOG:
  type: {component_type}
  category: {category}
  tags: [{tags}]
  location: {relative_path}

  provides: [{what_this_provides}]

"""
